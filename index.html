Should be quick to setup and throw away, plan for high IO usage! 
class gradle {

 version = 'gradle-1.0'

  // ... 

 archive { $version:
   ensure  => present,
   url   => "http://.../${version}-all.zip",
   checksum  => false,
   src_target  => '/tmp',
   target  => '/usr/share',
   extension => 'zip',
   require  => Package['unzip']
 }
}
